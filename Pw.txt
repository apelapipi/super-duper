function ConfigManager.LoadConfig()
    local success, result = pcall(function()
        if isfile("PetsWarOP_V2/config.json") then
            local data = readfile("PetsWarOP_V2/config.json")
            local decoded = Services.HttpService:JSONDecode(data)
            
            -- Deep merge configs
            for category, settings in pairs(decoded) do
                if Config[category] then
                    for key, value in pairs(settings) do
                        Config[category][key] = value
                    end
                end
            end
            
            return true
        end
        return false
    end)
    
    if success and result then
        WindUI:Notify({
            Title = "📂 Config Carregado",
            Content = "Configurações restauradas!",
            Duration = 3
        })
    end
end

local Utils = {}

-- Format number with dots (1000 -> 1.000)
function Utils.FormatNumberDots(number)
    number = math.floor(number or 0)
    local formatted = tostring(number)
    local result = ""
    local length = #formatted
    
    for i = 1, length do
        result = result .. formatted:sub(i, i)
        local remaining = length - i
        if remaining > 0 and remaining % 3 == 0 then
            result = result .. "."
        end
    end
    
    return result
end

-- Format number shortened (1000 -> 1K, 1000000 -> 1M)
function Utils.FormatNumberShort(number)
    number = number or 0
    
    if number >= 1e12 then
        return string.format("%.1fT", number / 1e12)
    elseif number >= 1e9 then
        return string.format("%.1fB", number / 1e9)
    elseif number >= 1e6 then
        return string.format("%.1fM", number / 1e6)
    elseif number >= 1e3 then
        return string.format("%.1fK", number / 1e3)
    end
    
    return tostring(math.floor(number))
end

-- HTTP Request wrapper
function Utils.HttpRequest(url)
    local requestFunc = (syn and syn.request) or request or http_request
    
    local success, response = pcall(requestFunc, {
        Url = url,
        Method = "GET"
    })
    
    if success and response and response.Body then
        local jsonSuccess, data = pcall(function()
            return Services.HttpService:JSONDecode(response.Body)
        end)
        
        if jsonSuccess then
            return data
        end
    end
    
    return nil
end

-- Get FPS color based on performance
function Utils.GetFPSColor(fps)
    if fps >= 55 then
        return "#44FF44"
    elseif fps >= 30 then
        return "#FFCC44"
    else
        return "#FF4444"
    end
end

-- Safe character check
function Utils.GetCharacterSafe(player)
    if not player then return nil end
    local character = player.Character
    if not character then return nil end
    
    local humanoid = character:FindFirstChild("Humanoid")
    local rootPart = character:FindFirstChild("HumanoidRootPart")
    
    if humanoid and rootPart and humanoid.Health > 0 then
        return character, humanoid, rootPart
    end
    
    return nil
end

-- Distance calculation
function Utils.GetDistance(pos1, pos2)
    if not pos1 or not pos2 then return math.huge end
    return (pos1 - pos2).Magnitude
end

function CombatSystem.Initialize()
    -- Create aura visualization sphere
    local sphere = Instance.new("Part")
    sphere.Name = "PW_AuraSphere_V2"
    sphere.Anchored = true
    sphere.CanCollide = false
    sphere.CanQuery = false
    sphere.CanTouch = false
    sphere.CastShadow = false
    sphere.Transparency = 0.75
    sphere.BrickColor = BrickColor.new("Medium lavender")
    sphere.Material = Enum.Material.Neon
    
    local mesh = Instance.new("SpecialMesh", sphere)
    mesh.MeshType = Enum.MeshType.Sphere
    
    CombatSystem.AuraPart = sphere
end

function CombatSystem.GetValidTargets()
    local targets = {}
    local _, _, playerRoot = Utils.GetCharacterSafe(Player)
    
    if not playerRoot then return targets end
    
    for _, mob in pairs(Services.CollectionService:GetTagged("Monster")) do
        local mobRoot = mob:FindFirstChild("HumanoidRootPart")
        local mobHumanoid = mob:FindFirstChildOfClass("Humanoid")
        
        if mobRoot and mobHumanoid and mobHumanoid.Health > 0 then
            local distance = Utils.GetDistance(playerRoot.Position, mobRoot.Position)
            
            if distance <= Config.Combat.AuraDistance then
                -- Check target mode
                local isValid = false
                
                if Config.Combat.TargetMode == "All" then
                    isValid = true
                elseif Config.Combat.TargetMode == "Custom" then
                    isValid = Config.Combat.SelectedMobs[mob.Name] == true
                elseif Config.Combat.TargetMode == "Nearest" then
                    isValid = #targets == 0 -- Only add first (nearest will be sorted)
                end
                
                if isValid then
                    table.insert(targets, {
                        Mob = mob,
                        Root = mobRoot,
                        Humanoid = mobHumanoid,
                        Distance = distance
                    })
                end
            end
        end
    end
    
    -- Sort by distance (nearest first)
    table.sort(targets, function(a, b)
        return a.Distance < b.Distance
    end)
    
    return targets
end

local MovementSystem = {}
MovementSystem.FlyBodyVelocity = nil
MovementSystem.FlyBodyGyro = nil
MovementSystem.FreecamPart = nil
MovementSystem.FlyUD = 0
MovementSystem.OriginalWalkSpeed = 16
MovementSystem.OriginalJumpPower = 50

function MovementSystem.Initialize()
    local char = Player.Character
    if char then
        local humanoid = char:FindFirstChild("Humanoid")
        if humanoid then
            MovementSystem.OriginalWalkSpeed = humanoid.WalkSpeed
            MovementSystem.OriginalJumpPower = humanoid.JumpPower or humanoid.JumpHeight
        end
    end
    
    -- Create freecam part
    local part = Instance.new("Part")
    part.Anchored = true
    part.CanCollide = false
    part.Transparency = 1
    MovementSystem.FreecamPart = part
end

function MovementSystem.UpdateFly()
    if not Config.Movement.Fly then
        MovementSystem.DisableFly()
        return
    end
    
    local char, humanoid, rootPart = Utils.GetCharacterSafe(Player)
    if not char or not humanoid or not rootPart then return end
    
    humanoid.PlatformStand = true
    
    -- Create body movers if they don't exist
    if not rootPart:FindFirstChild("FlyBodyVelocity") then
        local bv = Instance.new("BodyVelocity")
        bv.Name = "FlyBodyVelocity"
        bv.MaxForce = Vector3.new(9e9, 9e9, 9e9)
        bv.Parent = rootPart
        MovementSystem.FlyBodyVelocity = bv
    end
    
    if not rootPart:FindFirstChild("FlyBodyGyro") then
        local bg = Instance.new("BodyGyro")
        bg.Name = "FlyBodyGyro"
        bg.MaxTorque = Vector3.new(9e9, 9e9, 9e9)
        bg.P = 9e4
        bg.Parent = rootPart
        MovementSystem.FlyBodyGyro = bg
    end
    
    -- Update movement
    local moveVector = Controls:GetMoveVector()
    local lookVector = Camera.CFrame.LookVector
    local rightVector = Camera.CFrame.RightVector
    
    rootPart.FlyBodyGyro.CFrame = Camera.CFrame
    rootPart.FlyBodyVelocity.Velocity = (
        lookVector * -moveVector.Z + 
        rightVector * moveVector.X + 
        Vector3.new(0, MovementSystem.FlyUD, 0)
    ) * Config.Movement.FlySpeed
end

function MovementSystem.DisableFly()
    local char = Player.Character
    if char then
        local rootPart = char:FindFirstChild("HumanoidRootPart")
        local humanoid = char:FindFirstChild("Humanoid")
        
        if humanoid then
            humanoid.PlatformStand = false
        end
        
        if rootPart then
            local bv = rootPart:FindFirstChild("FlyBodyVelocity")
            local bg = rootPart:FindFirstChild("FlyBodyGyro")
            
            if bv then bv:Destroy() end
            if bg then bg:Destroy() end
        end
    end
    
    MovementSystem.FlyBodyVelocity = nil
    MovementSystem.FlyBodyGyro = nil
end

function MovementSystem.UpdateFreecam()
    if not Config.Movement.Freecam then
        MovementSystem.DisableFreecam()
        return
    end
    
    if MovementSystem.FreecamPart and MovementSystem.FreecamPart.Parent then
        local moveVector = Controls:GetMoveVector()
        local position = MovementSystem.FreecamPart.Position
        
        position = position + Camera.CFrame.LookVector * -moveVector.Z * Config.Movement.FreecamSpeed
        position = position + Camera.CFrame.RightVector * moveVector.X * Config.Movement.FreecamSpeed
        
        MovementSystem.FreecamPart.Position = position
        Camera.CFrame = CFrame.new(position, position + Camera.CFrame.LookVector)
    else
        local char = Player.Character
        if char then
            local rootPart = char:FindFirstChild("HumanoidRootPart")
            if rootPart then
                MovementSystem.FreecamPart.Position = rootPart.Position
                MovementSystem.FreecamPart.Parent = workspace
                Camera.CameraSubject = MovementSystem.FreecamPart
            end
        end
    end
end

function MovementSystem.DisableFreecam()
    if MovementSystem.FreecamPart then
        MovementSystem.FreecamPart.Parent = nil
    end
    
    local char = Player.Character
    if char then
        local humanoid = char:FindFirstChild("Humanoid")
        if humanoid then
            Camera.CameraSubject = humanoid
        end
    end
end

function MovementSystem.UpdateNoclip()
    if not Config.Movement.Noclip then return end
    
    local char = Player.Character
    if char then
        for _, part in pairs(char:GetDescendants()) do
            if part:IsA("BasePart") then
                part.CanCollide = false
            end
        end
    end
end

function MovementSystem.UpdateWalkSpeed()
    local char, humanoid = Utils.GetCharacterSafe(Player)
    if char and humanoid then
        humanoid.WalkSpeed = Config.Movement.WalkSpeed
        
        if Config.Movement.SpeedBoost then
            humanoid.WalkSpeed = Config.Movement.WalkSpeed * Config.Movement.SpeedMultiplier
        end
    end
end

function MovementSystem.UpdateJumpPower()
    local char, humanoid = Utils.GetCharacterSafe(Player)
    if char and humanoid then
        if humanoid.UseJumpPower then
            humanoid.JumpPower = Config.Movement.JumpPower
        else
            humanoid.JumpHeight = Config.Movement.JumpHeight
        end
    end
end

    -- Configure name
    esp.Name.Size = 14
    esp.Name.Center = true
    esp.Name.Outline = true
    esp.Name.Color = Color3.new(1, 1, 1)
    
    -- Configure box
    esp.Box.Thickness = 2

    esp.Box.Color = Color3.new(1, 1, 1)
    
    -- Configure line
    esp.Line.Thickness = 1
    esp.Line.Color = Color3.new(1, 1, 1)
    
    -- Configure distance
    esp.Distance.Size = 12
    esp.Distance.Center = true
    esp.Distance.Outline = true
    esp.Distance.Color = Color3.fromRGB(160, 160, 255)
    
    -- Configure health
    esp.Health.Size = 12
    esp.Health.Center = true
    esp.Health.Outline = true
    
    ESPSystem.PlayerESP[player] = esp
end

function ESPSystem.RemovePlayerESP(player)
    local esp = ESPSystem.PlayerESP[player]
    if esp then
        esp.Name:Remove()
        esp.Box:Remove()
        esp.Line:Remove()
        esp.Distance:Remove()
        esp.Health:Remove()
        ESPSystem.PlayerESP[player] = nil
    end
end

function ESPSystem.UpdatePlayerESP()
    for player, esp in pairs(ESPSystem.PlayerESP) do
        local char, humanoid, rootPart = Utils.GetCharacterSafe(player)
        
        if char and humanoid and rootPart then
            local head = char:FindFirstChild("Head") or rootPart
            local topPosition = head.Position + Vector3.new(0, 0.5, 0)
            local bottomPosition = rootPart.Position - Vector3.new(0, 3, 0)
            
            local topScreen, topVisible = Camera:WorldToViewportPoint(topPosition)
            local bottomScreen, bottomVisible = Camera:WorldToViewportPoint(bottomPosition)
            
            local color = Config.ESP.RandomColors and esp.Color or Color3.new(1, 1, 1)
            
            if topVisible or bottomVisible then
                local height = math.abs(topScreen.Y - bottomScreen.Y)
                local width = height / 2
                
                -- Update name
                esp.Name.Visible = Config.ESP.PlayerName
                esp.Name.Position = Vector2.new(topScreen.X, topScreen.Y - 18)
                esp.Name.Text = player.Name
                esp.Name.Color = color
                
                -- Update box
                esp.Box.Visible = Config.ESP.PlayerBox
                esp.Box.Size = Vector2.new(width, height)
                esp.Box.Position = Vector2.new(topScreen.X - width / 2, topScreen.Y)
                esp.Box.Color = color
                
                -- Update line
                esp.Line.Visible = Config.ESP.PlayerLine
                esp.Line.From = Vector2.new(Camera.ViewportSize.X / 2, Camera.ViewportSize.Y)
                esp.Line.To = Vector2.new(bottomScreen.X, bottomScreen.Y)
                esp.Line.Color = color
                
                -- Update distance
                if Config.ESP.PlayerDistance then
                    local _, _, playerRoot = Utils.GetCharacterSafe(Player)
                    if playerRoot then
                        local distance = math.floor(Utils.GetDistance(playerRoot.Position, rootPart.Position))
                        esp.Distance.Visible = true
                        esp.Distance.Position = Vector2.new(topScreen.X, topScreen.Y - 32)
                        esp.Distance.Text = distance .. "m"
                    else
                        esp.Distance.Visible = false
                    end
                else
                    esp.Distance.Visible = false
                end
                
                -- Update health
                if Config.ESP.PlayerHealth then
                    local healthRatio = math.clamp(humanoid.Health / math.max(humanoid.MaxHealth, 1), 0, 1)
                    esp.Health.Visible = true
                    esp.Health.Color = Color3.fromRGB(
                        math.floor(255 * (1 - healthRatio)),
                        math.floor(255 * healthRatio),
                        30
                    )
                    esp.Health.Position = Vector2.new(topScreen.X, bottomScreen.Y + 3)
                    esp.Health.Text = math.floor(healthRatio * 100) .. "%"
                else
                    esp.Health.Visible = false
                end
            else
                -- Hide all if not visible
                esp.Name.Visible = false
                esp.Box.Visible = false
                esp.Line.Visible = false
                esp.Distance.Visible = false
                esp.Health.Visible = false
            end
        else
            -- Hide all if character invalid
            esp.Name.Visible = false
            esp.Box.Visible = false
            esp.Line.Visible = false
            esp.Distance.Visible = false
            esp.Health.Visible = false
        end
    end
end

function ESPSystem.UpdateMobESP()
    -- Clean up dead mobs
    for mob, esp in pairs(ESPSystem.MobESP) do
        if not mob.Parent or not mob:FindFirstChild("HumanoidRootPart") then
            esp.Name:Remove()
            esp.Health:Remove()
            esp.Distance:Remove()
            ESPSystem.MobESP[mob] = nil
        end
    end
    
    -- Update or create ESP for alive mobs
    for _, mob in pairs(Services.CollectionService:GetTagged("Monster")) do
        local mobRoot = mob:FindFirstChild("HumanoidRootPart")
        local mobHumanoid = mob:FindFirstChildOfClass("Humanoid")
        
        if mobRoot and mobHumanoid and mobHumanoid.Health > 0 then
            -- Create ESP if doesn't exist
            if not ESPSystem.MobESP[mob] then
                ESPSystem.MobESP[mob] = {
                    Name = Drawing.new("Text"),
                    Health = Drawing.new("Text"),
                    Distance = Drawing.new("Text")
                }
                
                local esp = ESPSystem.MobESP[mob]
                esp.Name.Size = 14
                esp.Name.Center = true
                esp.Name.Outline = true
                esp.Name.Color = Color3.fromRGB(255, 95, 75)
                
                esp.Health.Size = 12
                esp.Health.Center = true
                esp.Health.Outline = true
                esp.Health.Color = Color3.fromRGB(255, 195, 45)
                
                esp.Distance.Size = 11
                esp.Distance.Center = true
                esp.Distance.Outline = true
                esp.Distance.Color = Color3.fromRGB(160, 160, 255)
            end
            
            local esp = ESPSystem.MobESP[mob]
            local screenPos, onScreen = Camera:WorldToViewportPoint(mobRoot.Position + Vector3.new(0, 4, 0))
            
            esp.Name.Visible = Config.ESP.MobEnabled and onScreen
            esp.Health.Visible = Config.ESP.MobHealth and onScreen
            esp.Distance.Visible = Config.ESP.MobDistance and onScreen
            
            if onScreen then
                esp.Name.Position = Vector2.new(screenPos.X, screenPos.Y)
                esp.Name.Text = mob.Name
                
                esp.Health.Position = Vector2.new(screenPos.X, screenPos.Y + 16)
                esp.Health.Text = Utils.FormatNumberShort(mobHumanoid.Health) .. "/" .. Utils.FormatNumberShort(mobHumanoid.MaxHealth)
                
                local _, _, playerRoot = Utils.GetCharacterSafe(Player)
                if playerRoot then
                    local distance = math.floor(Utils.GetDistance(playerRoot.Position, mobRoot.Position))
                    esp.Distance.Position = Vector2.new(screenPos.X, screenPos.Y - 16)
                    esp.Distance.Text = distance .. "m"
                end
            end
        end
    end
end

function OptimizationSystem.RemoveTextures(enabled)
    for _, obj in pairs(workspace:GetDescendants()) do
        if obj:IsA("BasePart") then
            if enabled then
                if not OptimizationSystem.OriginalProperties[obj] then
                    OptimizationSystem.OriginalProperties[obj] = {}
                end
                OptimizationSystem.OriginalProperties[obj].Material = obj.Material
                obj.Material = Enum.Material.SmoothPlastic
            elseif OptimizationSystem.OriginalProperties[obj] and OptimizationSystem.OriginalProperties[obj].Material then
                obj.Material = OptimizationSystem.OriginalProperties[obj].Material
            end
        end
    end
end

        if obj:IsA("ParticleEmitter") or obj:IsA("Trail") or obj:IsA("Smoke") or obj:IsA("Fire") or obj:IsA("Sparkles") then
            obj.Enabled = not enabled
        end
    end
end

                Content = player.DisplayName .. " (" .. player.Name .. ") entrou no servidor!",
                Duration = 10
            })

        end

function ServerUtils.ServerHop(sortOrder)
    pcall(function()
        local url = "https://games.roblox.com/v1/games/" .. game.PlaceId .. "/servers/Public?sortOrder=" .. sortOrder .. "&limit=100"
        local data = Utils.HttpRequest(url)
        
        if not data or not data.data then
            WindUI:Notify({
                Title = "❌ Erro",
                Content = "Não foi possível obter lista de servidores",
                Duration = 4
            })
            return
        end
        
        local validServers = {}
        for _, server in pairs(data.data) do
            if server.playing < server.maxPlayers and server.id ~= game.JobId then
                table.insert(validServers, server)
            end
        end
        
        if #validServers > 0 then
            local targetServer = validServers[1]
            if sortOrder == "Asc" then
                -- Find emptiest
                for _, server in ipairs(validServers) do
                    if server.playing < targetServer.playing then
                        targetServer = server
                    end
                end
            elseif sortOrder == "Desc" then
                -- Find fullest
                for _, server in ipairs(validServers) do
                    if server.playing > targetServer.playing then
                        targetServer = server
                    end
                end
            else
                -- Random
                targetServer = validServers[math.random(1, #validServers)]
            end
            
            WindUI:Notify({
                Title = "🌐 Server Hop",
                Content = "Conectando... (" .. targetServer.playing .. " jogadores)",
                Duration = 3
            })
            
            task.wait(0.5)
            Services.TeleportService:TeleportToPlaceInstance(game.PlaceId, targetServer.id, Player)
        else
            WindUI:Notify({
                Title = "❌ Erro",
                Content = "Nenhum servidor disponível",
                Duration = 4
            })
        end
    end)
end

-- 🎯 HITBOX SYSTEM

local HitboxSystem = {}
HitboxSystem.OriginalSizes = {}
HitboxSystem.DefaultHumanoidSize = Vector3.new(2, 2, 1)

function HitboxSystem.UpdatePlayerHitboxes()
    for _, player in pairs(Services.Players:GetPlayers()) do
        if player ~= Player then
            local char, _, rootPart = Utils.GetCharacterSafe(player)
            if char and rootPart then
                if Config.Hitbox.PlayerEnabled then
                    if not HitboxSystem.OriginalSizes[rootPart] then
                        HitboxSystem.OriginalSizes[rootPart] = rootPart.Size
                    end
                    
                    rootPart.Size = Vector3.new(
                        Config.Hitbox.PlayerSize,
                        Config.Hitbox.PlayerSize,
                        Config.Hitbox.PlayerSize
                    )
                    rootPart.Transparency = 0.7
                    rootPart.CanCollide = false
                    rootPart.CanQuery = false
                else
                    if HitboxSystem.OriginalSizes[rootPart] then
                        rootPart.Size = HitboxSystem.OriginalSizes[rootPart]
                        rootPart.Transparency = 1
                    else
                        rootPart.Size = HitboxSystem.DefaultHumanoidSize
                        rootPart.Transparency = 1
                    end
                end
            end
        end
    end
end

function HitboxSystem.UpdateMobHitboxes()
    for _, mob in pairs(Services.CollectionService:GetTagged("Monster")) do
        local mobRoot = mob:FindFirstChild("HumanoidRootPart")
        local mobHumanoid = mob:FindFirstChildOfClass("Humanoid")
        
        if mobRoot and mobHumanoid and mobHumanoid.Health > 0 then
            if Config.Hitbox.MobEnabled then
                if not HitboxSystem.OriginalSizes[mobRoot] then
                    HitboxSystem.OriginalSizes[mobRoot] = mobRoot.Size
                end
                
                mobRoot.Size = Vector3.new(
                    Config.Hitbox.MobSize,
                    Config.Hitbox.MobSize,
                    Config.Hitbox.MobSize
                )
                mobRoot.CanCollide = false
                mobRoot.CanQuery = false
                mobRoot.Transparency = 0.75
            else
                if HitboxSystem.OriginalSizes[mobRoot] then
                    mobRoot.Size = HitboxSystem.OriginalSizes[mobRoot]
                    mobRoot.Transparency = 0
                end
            end
        end
    end
end

-- 🔄 MAIN LOOP - ULTRA OPTIMIZED
-- ═══════════════════════════════════════════════════════════════════════════════
local lastUpdate = {
    Combat = 0,
    ESP = 0,
    Hitbox = 0,
    Stats = 0,
}

local updateIntervals = {
    Combat = 0.05,  -- 20 times per second
    ESP = 0.033,    -- 30 times per second
    Hitbox = 0.1,   -- 10 times per second
    Stats = 0.3,    -- ~3 times per second
}

Services.RunService.RenderStepped:Connect(function(deltaTime)
    local currentTime = tick()
    
    -- Combat System
    if currentTime - lastUpdate.Combat >= updateIntervals.Combat then
        CombatSystem.PerformKillAura()
        CombatSystem.UpdateAuraVisual()
        lastUpdate.Combat = currentTime
    end
    
    -- ESP System
    if currentTime - lastUpdate.ESP >= updateIntervals.ESP then
        ESPSystem.UpdatePlayerESP()
        ESPSystem.UpdateMobESP()
        lastUpdate.ESP = currentTime
    end
    
    -- Hitbox System
    if currentTime - lastUpdate.Hitbox >= updateIntervals.Hitbox then
        HitboxSystem.UpdatePlayerHitboxes()
        HitboxSystem.UpdateMobHitboxes()
        lastUpdate.Hitbox = currentTime
    end
    
    -- Stats Update
    if currentTime - lastUpdate.Stats >= updateIntervals.Stats then
        VisualSystem.UpdateStats(deltaTime)
        lastUpdate.Stats = currentTime
    end
    
    -- Movement System (needs to run every frame for smooth fly)
    MovementSystem.UpdateFly()
    MovementSystem.UpdateFreecam()
    MovementSystem.UpdateNoclip()
    
    -- Auto-save config
    if Config.System.AutoSave and currentTime % 60 < deltaTime then
        ConfigManager.SaveConfig()
    end
    
    -- AFK prevention
    if Config.Movement.AFKMode and currentTime % 300 < deltaTime then
        Services.VirtualUser:Button1Down(Vector2.new(0, 0))
        task.wait(0.1)
        Services.VirtualUser:Button1Up(Vector2.new(0, 0))
    end
end)
